        void about() const noexcept {
            std::cout << "I'm " << _name;
            if(!_lang.empty()) std::cout << ", a " << _lang << " programmer";
            std::cout << " =)" << std::endl;
        }
    private:
        std::string _name{};
        std::string _lang{};
    };
}

int main() {
    gtd::Speaker speaker{"Dany","C++"};
    speaker.hello("GitHub");
    speaker.about();
    return 0;
}
